  def port_configuration(rule_name, info)

    # Get to the advanced page.

    self.goto_advanced(rule_name, info)
    
    # Get to the "Port Configuration" page.

    begin
      @ff.link(:text, 'Port Configuration').click
      self.msg(rule_name, :info, 'Port Configuration', 'Reached page \'Port Configuration\'.')
    rescue
      self.msg(rule_name, :error, 'Port Configuration', 'Did not reach \'Port Configuration\' page')
      return
    end
    
    # Check the key.

    if ( info.has_key?('section') &&
         info.has_key?('subsection') ) then
      # Right,go on.

    else
      self.msg(rule_name,:error,'local_administration','Some key NOT found.')
      return
    end 
    
    # parse the json file.

    
    # "Port1"

    if info.has_key?('Port1')
      
      case info['Port1']
      
      when 'Auto'
        
        # Set "Auto"

        @ff.select_list(:name,'port_name_0').select_value("0")
        self.msg(rule_name,:info,'Port1',info['Port1'])
 
      when '10 Half Duplex'
        
        # Set "10 Half Duplex"

        @ff.select_list(:name,'port_name_0').select_value("1")
        self.msg(rule_name,:info,'Port1',info['Port1'])
        
      when '10 Full Duplex'
        
        # Set "10 Full Duplex"

        @ff.select_list(:name,'port_name_0').select_value("2")
        self.msg(rule_name,:info,'Port1',info['Port1'])
        
      when '100 Half Duplex'
        
        # Set "100 Half Duplex"

        @ff.select_list(:name,'port_name_0').select_value("3")
        self.msg(rule_name,:info,'Port1',info['Port1'])
               
      when '100 Full Deplex'
        
        # Set "100 Full Duplex"

        @ff.select_list(:name,'port_name_0').select_value("4")
        self.msg(rule_name,:info,'Port1',info['Port1'])
        
      else
        
        # Wrong here

        self.msg(rule_name,:error,'port_configuration','Did NOT find the value in \'Port1\'.')
        return
        
      end # end of case

      
    end # end of if  


    # "Port2"

    if info.has_key?('Port2')
      
      case info['Port2']
      
      when 'Auto'
        
        # Set "Auto"

        @ff.select_list(:name,'port_name_1').select_value("0")
        self.msg(rule_name,:info,'Port2',info['Port2'])
 
      when '10 Half Duplex'
        
        # Set "10 Half Duplex"

        @ff.select_list(:name,'port_name_1').select_value("1")
        self.msg(rule_name,:info,'Port2',info['Port2'])
        
      when '10 Full Duplex'
        
        # Set "10 Full Duplex"

        @ff.select_list(:name,'port_name_1').select_value("2")
        self.msg(rule_name,:info,'Port2',info['Port2'])
        
      when '100 Half Duplex'
        
        # Set "100 Half Duplex"

        @ff.select_list(:name,'port_name_1').select_value("3")
        self.msg(rule_name,:info,'Port2',info['Port2'])
               
      when '100 Full Deplex'
        
        # Set "100 Full Duplex"

        @ff.select_list(:name,'port_name_1').select_value("4")
        self.msg(rule_name,:info,'Port2',info['Port2'])
        
      else
        
        # Wrong here

        self.msg(rule_name,:error,'port_configuration','Did NOT find the value in \'Port2\'.')
        return
        
      end # end of case

      
    end # end of if    

    
    # "Port3"

    if info.has_key?('Port3')
      
      case info['Port3']
      
      when 'Auto'
        
        # Set "Auto"

        @ff.select_list(:name,'port_name_2').select_value("0")
        self.msg(rule_name,:info,'Port3',info['Port3'])
 
      when '10 Half Duplex'
        
        # Set "10 Half Duplex"

        @ff.select_list(:name,'port_name_2').select_value("1")
        self.msg(rule_name,:info,'Port3',info['Port3'])
        
      when '10 Full Duplex'
        
        # Set "10 Full Duplex"

        @ff.select_list(:name,'port_name_2').select_value("2")
        self.msg(rule_name,:info,'Port3',info['Port3'])
        
      when '100 Half Duplex'
        
        # Set "100 Half Duplex"

        @ff.select_list(:name,'port_name_2').select_value("3")
        self.msg(rule_name,:info,'Port3',info['Port3'])
               
      when '100 Full Deplex'
        
        # Set "100 Full Duplex"

        @ff.select_list(:name,'port_name_2').select_value("4")
        self.msg(rule_name,:info,'Port3',info['Port3'])
        
      else
        
        # Wrong here

        self.msg(rule_name,:error,'port_configuration','Did NOT find the value in \'Port3\'.')
        return
        
      end # end of case

      
    end # end of if 


    # "Port4"

    if info.has_key?('Port4')
      
      case info['Port4']
      
      when 'Auto'
        
        # Set "Auto"

        @ff.select_list(:name,'port_name_3').select_value("0")
        self.msg(rule_name,:info,'Port4',info['Port4'])
 
      when '10 Half Duplex'
        
        # Set "10 Half Duplex"

        @ff.select_list(:name,'port_name_3').select_value("1")
        self.msg(rule_name,:info,'Port4',info['Port4'])
        
      when '10 Full Duplex'
        
        # Set "10 Full Duplex"

        @ff.select_list(:name,'port_name_3').select_value("2")
        self.msg(rule_name,:info,'Port4',info['Port4'])
        
      when '100 Half Duplex'
        
        # Set "100 Half Duplex"

        @ff.select_list(:name,'port_name_3').select_value("3")
        self.msg(rule_name,:info,'Port4',info['Port4'])
               
      when '100 Full Deplex'
        
        # Set "100 Full Duplex"

        @ff.select_list(:name,'port_name_3').select_value("4")
        self.msg(rule_name,:info,'Port4',info['Port4'])
        
      else
        
        # Wrong here

        self.msg(rule_name,:error,'port_configuration','Did NOT find the value in \'Port4\'.')
        return
        
      end # end of case

      
    end # end of if  

    
    # Apply for the change

    @ff.link(:text,'Apply').click
    
    # Making change will cause the "Attention" page.

    if @ff.text.include?'Attention'
      @ff.link(:text,'Apply').click
      
      # Wait for 30 seconds

      sleep 30
    end

    # Output the result

    self.msg(rule_name,:info,'Port Configuration','SUCCESS')
    
  end
