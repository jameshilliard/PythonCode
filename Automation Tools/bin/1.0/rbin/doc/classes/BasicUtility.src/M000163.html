  def logout(rule_name, info)
    
    # jump to the main page

    self.main(rule_name, info)
    
    # click logout

    begin
      @ff.link(:text, 'Logout').click
      self.msg(rule_name, :info, 'logout', 'success')
    rescue
      self.msg(rule_name, :error, 'logout', 'failed')
    end
  end
