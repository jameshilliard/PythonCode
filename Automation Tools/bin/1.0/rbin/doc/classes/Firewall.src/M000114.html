  def StaticNAT(rule_name, info)
    
    if info.key?('page')
      
      case info['page']   
        
      when 'Static NAT'
        GoStaticNATPage(rule_name, info)
        DoSetup_StaticNAT(rule_name, info)
        
      else
        self.msg(rule_name, :error, 'StaticNAT()', 'page undefined')
        
      end # end of case

      
    else
      self.msg(rule_name, :error, 'StaticNAT()', 'No page key found')
      
    end # end of if

    
  end
