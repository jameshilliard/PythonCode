    if info.key?('layout')
      case info['layout']
      when 'Wireless Status'
        WirelessStatus(rule_name, info)
      when 'Basic Security Settings'
        BasicSecuritySettings(rule_name, info)
      when 'Advanced Security Settings'
        AdvancedSecuritySettings(rule_name, info)
